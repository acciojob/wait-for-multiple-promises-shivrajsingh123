document.addEventListener("DOMContentLoaded", function () { const outputTable = document.getElementById("output"); // Function to create a promise that resolves after a random delay between 1 and 3 seconds function createPromise(id) { const delay = Math.random() * 2000 + 1000; // Random delay between 1000ms (1s) and 3000ms (3s) return new Promise((resolve) => { setTimeout(() => { resolve({ id, time: (delay / 1000).toFixed(3) }); }, delay); }); } // Create three promises const promises = [createPromise(1), createPromise(2), createPromise(3)]; // Display the "Loading..." row initially outputTable.innerHTML = `Loading...`; // Wait for all promises to resolve Promise.all(promises).then(results => { // Remove the "Loading..." row outputTable.innerHTML = ""; // Populate the table with resolved values results.forEach(result => { const row = document.createElement("tr"); row.innerHTML = `Promise ${result.id}${result.time}`; outputTable.appendChild(row); }); // Calculate the maximum time taken (i.e., the longest resolving promise) const maxTime = Math.max(...results.map(r => parseFloat(r.time))).toFixed(3); // Add total row const totalRow = document.createElement("tr"); totalRow.innerHTML = `Total${maxTime}`; outputTable.appendChild(totalRow); }); });
